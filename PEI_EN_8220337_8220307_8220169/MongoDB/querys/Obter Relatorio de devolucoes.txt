use('BdPeiTP');

// Obter Relatório de devoluções para um determinado mês, de um determinado cliente
var primeiroNomeParceiro = "LAURA"; // Substitua pelo nome específico do parceiro
var ultimoNomeParceiro = "RODRIGUEZ"
var ano = 2022; // Substitua pelo ano desejado
var mes = 08; // Substitua pelo mês desejado (1 para janeiro, 2 para fevereiro, etc.)


db.returns.aggregate([
    {
        $match: {
            "customer.first_name": primeiroNomeParceiro,
            "customer.last_name": ultimoNomeParceiro,
            date: {
                $gte: new Date(ano, mes-1, 0),
                $lt: new Date(ano, mes, 0)
            }
        }
    },
    {
        $addFields: {
            // Para cada devolução deverá ser apresentado o número de dias que passaram até à devolução do produto
            daysUntilReturn: {
                $divide: [
                    { $subtract: ["$date", "$sale.date"] },
                    1000 * 60 * 60 * 24 // transforma em milisegundos
                ]
            },
            // Indicador de devolução precoce, isto é, se o produto foi ou não devolvido no prazo de 3 dias após a compra
            earlyReturn: {
                $cond: {
                    if: { $lt: ["$date", { $add: ["$sale.date", 3 * 24 * 60 * 60 * 1000] }] }, // transforma em milisegundos
                    then: true,
                    else: false
                }
            }
        }
    }
]);
